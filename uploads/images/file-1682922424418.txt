1. npm init
2. npm i 
3. apply first code 
	const express = require("express")
	var app = express()

	app.get("/",function(request,response){
		response.send("Hello World!")
	})

	app.listen(3000, function () {
		console.log("Started application on port %d", 3000)
	});
	
4. npm i --save express

	run project > node index.js

5. create routes  

	app.get("/home",function(request,response){
		response.send("Hello World!")
	})
	
	app.post("/loginData",function(request,response){
		response.send("Hello World!")
	})
	
	app.put("/update",function(request,response){
		response.send("Hello World!")
	})
	app.delete("/delete/user/:id",function(request,response){
		response.send("Hello World!")
	})


6. add nodemoon for watching(mean no rerun project)
	npm i nodemon
	
	add cmd in pacakge.json file
	
	 "scripts": {
		"test": "npm run test",
		"start":"nodemon index.js"   
	  },
	  
	run nodemon project watching 
	
		npm run start
